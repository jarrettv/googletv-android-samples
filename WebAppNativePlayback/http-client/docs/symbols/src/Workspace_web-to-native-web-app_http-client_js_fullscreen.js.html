 15 /**
 16  * The constructor for the fullscren video page. Sets up video playback
 17  *
 18  * @constructor
 19  * @param {VideoDataController} videoDataController
 20  * @param {FocusController} focusController
 21  */
 22 function FullscreenPage(vidDataController, fController) {
 23     // Set up variables
 24     var videoDataController = vidDataController;
 25     var focusController = fController;
 26     var categoryIndex = -1;
 27     var itemIndex = -1;
 28     var embedded = false;
 29     var videoPlayerController = null;
 30     var videoControlsController = null;
 31 
 32     /**
 33     * Get the selected category index
 34     */
 35     this.getCategoryIndex = function () {
 36         return categoryIndex;
 37     };
 38 
 39     /**
 40     * Set the selected category index
 41     * @param {int} index
 42     */
 43     this.setCategoryIndex = function (index) {
 44         categoryIndex = index;
 45     };
 46 
 47     /**
 48     * Get the selected item index
 49     */
 50     this.getItemIndex = function () {
 51         return itemIndex;
 52     };
 53 
 54     /**
 55     * Set the selected item index
 56     * @param {int} index
 57     */
 58     this.setItemIndex = function (index) {
 59         itemIndex = index;
 60     };
 61 
 62     /**
 63     * Set the embedded state
 64     * @param {boolean} isEmbedded
 65     */
 66     this.setEmbedded = function (isEmbedded) {
 67         embedded = isEmbedded;
 68     };
 69

 71     * Check whether the view is embedded
 72     */

 74         return embedded;
 75     };
 76 
 77     /**
 78     * Get the VideoDataController
 79     */
 80     this.getVideoDataController = function () {
 81         return videoDataController;
 82     };
 83 
 84     /**
 85     * Set the VideoPlayerController for the view
 86     * @param {VideoPlayerController} controller
 87     */
 88     this.setVideoPlayerController = function (controller) {
 89         videoPlayerController = controller;
 90     };
 91 
 92     /**
 93     * Get the VideoPlayerController
 94     */

 99     /**
100     * Set the VideoControlsController for the view

105     };

145 /**
146 * Set up the page state
147 * @function
148 */
149 FullscreenPage.prototype.setUpPageState = function () {
150     var queryString = location.search;
151 
152     if (queryString.length < 1) {
153         return;
154     }
155 
156     var parms = queryString.substring(1).split('&');
157     // We must have 2 or 3 parameters
158     if (parms.length < 2) {
159         return;
160     }
161 
162     var paramSplit;
163     var categoryIndex = -1;
164     var itemIndex = -1;
165     var embedded = false;
166 
167     var paramString;
168     for(var  i = 0; i < parms.length; i++) {
169         paramString = parms[i];
170         paramSplit = paramString.split("=");
171         if(paramSplit.length != 2) {
172             continue;
173         }
174 
175         if(paramSplit[0].toLowerCase() == 'category') {
176             categoryIndex = parseInt(paramSplit[1], 10);
177         } else if(paramSplit[0].toLowerCase() == 'item') {
178             itemIndex = parseInt(paramSplit[1], 10);
179         } else if(paramSplit[0].toLowerCase() == 'embedded') {
180             embedded = paramSplit[1].toLowerCase() == 'true';
181         }
182     }
183 
184     this.setCategoryIndex(categoryIndex);
185     this.setItemIndex(itemIndex);
186     this.setEmbedded(embedded);
187 };

189 /**
190 * Set the up video player
191 * @function
192 */
193 FullscreenPage.prototype.setUpVideoPlayer = function () {
194     var videoPlayerController = this.getVideoPlayerController();
195     var videoControlsController = this.getVideoControlsController();
196     var videoDataController = this.getVideoDataController();
197     var videoData = videoDataController.getVideoData(this.getCategoryIndex())[this.getItemIndex()];
198     videoPlayerController.setVideoUrl(videoData.src);
199     videoControlsController.reset();
200 };
201 
202 /**
203 * A callback for the handle video controls events
204 * @function
205 * @param {String} action An action defined in the VideoControlsController
206 */
207 FullscreenPage.prototype.onHandleVideoControlEventCommand = function (action) {
208     var videoPlayer = this.getVideoPlayerController();
209     if(!videoPlayer) {
210         return;
211     }
212 
213     if(action == ACTION_PLAY || action == ACTION_PAUSE) {
214         videoPlayer.playPause();
215     } else if(action == ACTION_REWIND) {
216         videoPlayer.rewind();
217     } else if(action == ACTION_FF) {
218         videoPlayer.fastForward();
219     } else if(action == ACTION_PREV) {
220         var prevItemIndex = this.getItemIndex() - 1;
221         if(prevItemIndex < 0) {
222             prevItemIndex = this.getVideoDataController().getVideoData(this.getCategoryIndex()).length - 1;
223         }
224         this.setItemIndex(prevItemIndex);
225         this.setUpVideoPlayer();
226         //location.assign('./fullscreen.html?category='+this.getCategoryIndex()+'&item='+prevItemIndex);
227     } else if(action == ACTION_NEXT) {
228         var nextItemIndex = this.getItemIndex() + 1;
229         if(nextItemIndex >= this.getVideoDataController().getVideoData(this.getCategoryIndex()).length) {
230             nextItemIndex = 0;
231         }
232         this.setItemIndex(nextItemIndex);
233         this.setUpVideoPlayer();
234         //location.assign('./fullscreen.html?category='+this.getCategoryIndex()+'&item='+nextItemIndex);
235     }
236 };

238 /**
239 * Method to initialise all the controllers used by this class
240 * @private
241 * @function
242 */
243 FullscreenPage.prototype.initControllers = function () {
244     var videoControlsController = this.getVideoControlsController();
245     videoControlsController.setOnVideoCommandCallback(this.getOnVideoControlEventCallback());
246 
247     var videoPlayerController = this.getVideoPlayerController();
248     videoPlayerController.setPlayerCallback(this.getOnVideoPlayerEventCallback());
249 };
250 
251 /**
252 * A callback method for a video playback event
253 * @function
254 * @param {String} type The type of event defined in VideoPlayerController
255 * @param {String} data Data relevant to the event type
256 */
257 FullscreenPage.prototype.onVideoPlayerEvent = function (type, data) {
258     if(type == VIDEO_EVENT_STATE_CHANGE) {
259         this.getVideoControlsController().setState(data);
260     } else if(type == VIDEO_EVENT_PROGRESS_UPDATE) {
261         this.getVideoControlsController().setLoadedTime(data);
262     } else if(type == VIDEO_EVENT_TIME_UPDATE) {
263         this.getVideoControlsController().setElapsedTime(data);
264     } else if(type == VIDEO_EVENT_DURATION_CHANGE) {
265         this.getVideoControlsController().setDuration(data);
266     } else if(type == VIDEO_EVENT_LOADED) {
267         this.getVideoControlsController().setDuration(data);
268     } else if(type == VIDEO_EVENT_MOUSE_MOVED) {
269         // NOOP
270     }
271 };

282 /**
283 * Handle URIs from the Native application to JS
284 * @function
285 * @param {Stirng} uri The Uri passed from native to the Javascript
286 */
287 FullscreenPage.prototype.handleUri = function (uri) {
288     if(uri.indexOf(this.URI_PREFIX) !== 0) {
289         // Invalid uri for us to handle
290         return;
291     }
292 
293     uri = uri.substring(this.URI_PREFIX.length, uri.length);
294 
295     var uriParts = uri.split(";");
296     if(!uriParts || !uriParts.length || uriParts.length === 0) {
297         return;
298     }
299 
300     var action = uriParts[0];
301     if(action == 'DATA_DURATION') {
302         this.getVideoControlsController().setDuration(uriParts[1]);
303     } else if(action == 'DATA_CURRENT_POSITION') {
304         this.getVideoControlsController().setElapsedTime(uriParts[1]);
305     } else if(action == 'DATA_BUFFERING_PERCENT') {
306         this.getVideoControlsController().setLoadedPercentage(uriParts[1]);
307     } else if(action == 'DATA_PLAY_STATE') {
308         var state = uriParts[1].toLowerCase();
309         if(state == 'playing') {
310             this.onVideoPlayerEvent(VIDEO_EVENT_STATE_CHANGE, STATE_PLAYING);
311         } else {
312             this.onVideoPlayerEvent(VIDEO_EVENT_STATE_CHANGE, STATE_PAUSED);
313         }
314 
315     }
316 };

318 /**
319 * Set up key handling if the view is embedded (i.e. in Android)
320 * @function
321 */
322 FullscreenPage.prototype.setUpKeyHandling = function () {
323     if(this.isEmbedded()) {
324       // We want the native app to access this
325       window.androidKeyHandler = new AndroidKeyHandler(this.getFocusController());
326   }

329 var fullscreenPage;
330 
331 $(document).ready(function() {
332     if(fullscreenPage !== undefined) {
333         return;
334     }
335 
336     var focusController = new FocusController();
337 
338     fullscreenPage = new FullscreenPage(new VideoDataController(), focusController);
339     fullscreenPage.setUpPageState();
340 
341     var videoPlayerController;
342     if(fullscreenPage.isEmbedded()) {
343         videoPlayerController = new NativeVideoPlayerController();
344         window.fullscreenPage = fullscreenPage;
345     } else {
346         videoPlayerController = new WebVideoPlayerController('video-container');
347     }
348 
349     fullscreenPage.setUpKeyHandling();
350 
351     fullscreenPage.setVideoPlayerController(videoPlayerController);
352     fullscreenPage.setVideoControlsController(new VideoControlsController('controls-container', focusController));
353     fullscreenPage.initControllers();
354     fullscreenPage.setUpVideoPlayer();
355 
356     fullscreenPage.initialiseFocus();
357 
358     $(document.body).css('visibility', 'visible');
359 });
