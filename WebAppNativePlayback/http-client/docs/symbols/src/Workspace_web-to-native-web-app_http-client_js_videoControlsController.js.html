 15 /**
 16  * The VideoControlsController handles the UI for the Video controls and gets
 17  * information from a GenericVideoPlayerController
 18  *
 19  * @constructor
 20  * @param {int} containerId Container Id for the controls to be placed
 21  * @param {FocusController} focusController Focus Controller for the buttons
 22  */
 23 function VideoControlsController(contId, fController) {
 24     var containerId = contId;
 25     var videoCommandCallback = null;
 26     var focusableItems = [];
 27     var focusController = fController;
 28     var loadedPercent = 0;
 29     var duration = 0;
 30     var elapsedTime = 0;
 31 
 32     this.CONTROL_CONTAINER_CLASS = 'video-command-container';
 33     this.VIDEO_COMMAND_CLASS = 'video-command';
 34     this.PROGRESS_BAR_ID = 'progress-bar';
 35     this.PROGRESS_BAR_CLASS = 'progress-bar';
 36     this.PROGRESS_TIME_CLASS = 'progress-time';
 37     this.PROGRESS_ELAPSED_TIME_CLASS = 'progress-elapsed-time';
 38     this.ELAPSED_TIME_FS_CLASS = 'elapsed-time-fs';
 39     this.ELAPSED_TIME_ID = 'elapsed-time';
 40     this.ELAPSED_TIME_CLASS = 'elapsed-time';
 41     this.PROGRESS_BAR_FS_CLASS = 'progressbar-fs';
 42     this.PROGRESS_BAR_BG_CLASS = 'progress-bar-back';
 43     this.PROGRESS_PLAY_CLASS = 'progress-play-progress';
 44     this.PROGRESS_TIME_FS_CLASS = 'progress-time-fs';
 45     this.PROGRESS_LOAD_CLASS = 'progress-load-progress';
 46     this.LOAD_TIME_FS_CLASS = 'load-time-fs';
 47     this.PROGRESS_TIME_CLASS = 'progress-time';
 48     this.PROGRESS_DURATION_CLASS = 'progress-duration';
 49     this.DURATION_FS_CLASS = 'duration-fs';
 50     this.DURATION_TIME_ID = 'duration-time';
 51     this.DURATION_TIME_CLASS = 'duration-time';
 52     this.PLAY_PAUSE_BUTTON_CLASS = 'play-pause-fs';
 53     this.FAST_FORWARD_BUTTON_CLASS = 'fast-forward-fs';
 54     this.REWIND_BUTTON_CLASS = 'rewind-fs';
 55     this.PREVIOUS_BUTTON_CLASS = 'previous-fs';
 56     this.NEXT_BUTTON_CLASS = 'next-fs';
 57 
 58     /**
 59     * Get the container Id
 60     */
 61     this.getContainerId = function () {
 62         return containerId;
 63     };
 64 
 65     /**
 66     * This method will handle calling the correct callback for a video command
 67     * @param {String} action
 68     */
 69     this.onVideoCommand = function (action) {
 70         var callback = this.getOnVideoCommandClickCallback();
 71         if(callback) {
 72             callback(action);
 73         }
 74     };
 75 
 76     /**
 77     * Add a focusable item to the focus controller
 78     * @param {FocusableItem} focusableItem
 79     */
 80     this.addFocusableItem = function (focusableItem) {
 81         focusableItems.push(focusableItem);
 82         focusController.addFocusableItem(focusableItem);
 83     };
 84 
 85     /**
 86     * Remove all the focusable items from the focus controller
 87     */
 88     this.removeFocusableItems = function () {
 89         for(var i = 0; i < focusableItems.length; i++) {
 90             focusController.removeFocusableItem(focusableItems[i]);
 91         }
 92     };
 93 
 94     /**
 95     * Set the callback method for video commands
 96     * @param {function} callback
 97     */
 98     this.setOnVideoCommandCallback = function (callback) {
 99         this.videoCommandCallback = callback;
100     };
101 
102     /**
103     *Get the video commands callback
104     */
105     this.getOnVideoCommandClickCallback = function () {
106         return this.videoCommandCallback;
107     };
108 
109     /**
110     * Set the state of the controller (Playing / Paused / Stopped)
111     * @param {String} state
112     */
113     this.setState = function (state) {
114         var image;
115         if(state == STATE_PLAYING) {
116             image = $($('#'+this.PLAY_PAUSE_BUTTON_CLASS+' img')[0]);
117             image.attr('src', '../images/bt-pause.png');
118         } else if(state == STATE_PAUSED || state == STATE_STOPPED) {
119             image = $($('#'+this.PLAY_PAUSE_BUTTON_CLASS+' img')[0]);
120             image.attr('src', '../images/bt-playFS.png');
121         }
122     };
123 
124     /**
125     * Get the duration of the video controls
126     */
127     this.getDuration = function () {
128         return this.duration;
129     };
130 
131     /**
132     * Set the duration of the controls
133     * @param {int} seconds
134     */
135     this.setDuration = function (seconds) {
136         this.duration = seconds;
137         $('#'+this.DURATION_TIME_ID).html(formatSeconds(seconds));
138         this.updateProgressBar();
139     };
140 
141     /**
142     * Set the elapsed time of playback
143     * @param {int} seconds
144     */
145     this.setElapsedTime = function (seconds) {
146         this.elapsedTime = seconds;
147         $('#'+this.ELAPSED_TIME_ID).html(formatSeconds(seconds));
148 
149         this.updateProgressBar();
150     };
151 
152     /**
153     * Get the elapsed time of playback
154     */
155     this.getElapsedTime = function () {
156         return this.elapsedTime;
157     };
158 
159     /**
160     * Update the progress bar UI
161     */
162     this.updateProgressBar = function () {
163         var progressPercent = Math.round(this.getElapsedTime() * 100 / this.getDuration());
164 
165         this.updateBufferedPercentage(this.getLoadedPercentage());
166         $('.'+this.PROGRESS_TIME_FS_CLASS).css('width', progressPercent+'%');
167     };
168 
169     /**
170     * Get the loaded percent
171     */
172     this.getLoadedPercentage = function () {
173         return loadedPercent;
174     };
175 
176     /**
177     * Set the loaded time of the video
178     * @param {int} seconds
179     */
180     this.setLoadedTime = function (seconds) {
181         var percent = Math.round(seconds * 100 / this.getDuration());
182         this.setLoadedPercentage(percent);
183     };
184 
185     /**
186     * Set the loaded percentage
187     * @param {int} percent
188     */
189     this.setLoadedPercentage = function (percent) {
190         loadedPercent = percent;
191         this.updateProgressBar();
192     };
193 
194     /**
195     * Update the buffered percentage
196     * @param {int} percent
197     */
198     this.updateBufferedPercentage = function (percent) {
199         $('.'+this.LOAD_TIME_FS_CLASS).css('width', percent + '%');
200     };
201 
202     /**
203     * Reset the controls state
204     */
205     this.reset = function () {
206         this.setState(STATE_STOPPED);
207         this.setElapsedTime(0);
208         this.setLoadedTime(0);
209         this.setDuration(0);
210     };
211 
212     appendHTMLMarkup.bind(this)();
213 
214     /**
215     * Add the HTML markup to the container Id
216     * @private
217     */
218     function appendHTMLMarkup() {
219         if(!this.getContainerId()) {
220             // No ContainerId
221             console.log("VideoControlsController: No Contianer ID");
222             return;
223         }
224 
225         var controlsContainer = $('#'+this.getContainerId());
226 
227         // First by adding play, pause, rewind etc
228         var videoCommands = getVideoCommands.bind(this)();
229         var focusableItem;
230         for(var i = 0; i < videoCommands.length; i++) {
231             var container = $('<div></div>').addClass(this.CONTROL_CONTAINER_CLASS);
232 
233             var command = $('<div></div>').addClass(this.VIDEO_COMMAND_CLASS);
234             command.attr('id', videoCommands[i].id);
235 
236             // Make command focusable here
237             focusableItem = new VideoCommandFocusableItem(command, videoCommands[i].action);
238             focusableItem.setOnItemClickCallback(this.onVideoCommand.bind(this));
239             this.addFocusableItem(focusableItem);
240 
241             var commandImg = $('<img />').attr('src', videoCommands[i].url);
242 
243             command.append(commandImg);
244             container.append(command);
245             controlsContainer.append(container);
246         }
247 
248         // Add progress bar
249         var progressBar = $("<div></div>").attr('id', this.PROGRESS_BAR_ID).addClass(this.PROGRESS_BAR_CLASS);
250 
251         var elapsedTime = $('<div></div>');
252         elapsedTime.addClass(this.PROGRESS_TIME_CLASS);
253         elapsedTime.addClass(this.PROGRESS_ELAPSED_TIME_CLASS);
254         elapsedTime.addClass(this.ELAPSED_TIME_FS_CLASS);
255 
256         var elapsedSpan = $('<span></span>').attr('id', this.ELAPSED_TIME_ID).addClass(this.ELAPSED_TIME_CLASS);
257         elapsedTime.append(elapsedSpan);
258 
259         progressBar.append(elapsedTime);
260 
261         var videoTimeline = $('<div></div>').addClass(this.PROGRESS_BAR_FS_CLASS);
262         var timelineBack = $('<div></div>').addClass(this.PROGRESS_BAR_BG_CLASS);
263 
264         var progressTime = $('<div></div>').addClass(this.PROGRESS_PLAY_CLASS);
265         progressTime.addClass(this.PROGRESS_TIME_FS_CLASS);
266         timelineBack.append(progressTime);
267 
268         var loadTime = $('<div</div>').addClass(this.PROGRESS_LOAD_CLASS);
269         loadTime.addClass(this.LOAD_TIME_FS_CLASS);
270         timelineBack.append(loadTime);
271 
272         videoTimeline.append(timelineBack);
273 
274         progressBar.append(videoTimeline);
275 
276         var durationTime = $('<div></div>');
277         durationTime.addClass(this.PROGRESS_TIME_CLASS);
278         durationTime.addClass(this.PROGRESS_DURATION_CLASS);
279         durationTime.addClass(this.DURATION_FS_CLASS);
280 
281         var durationSpan = $('<span></span>').attr('id', this.DURATION_TIME_ID).addClass(this.DURATION_TIME_CLASS);
282         durationTime.append(durationSpan);
283 
284         progressBar.append(durationTime);
285 
286         controlsContainer.append(progressBar);
287     }

289     /**
290     * Get an array of video commands for the video playback
291     * @private
292     */
293     function getVideoCommands() {
294         return [
295             getVideoCommand(this.PREVIOUS_BUTTON_CLASS, '../images/bt-previousFS.png', ACTION_PREV),
296             getVideoCommand(this.REWIND_BUTTON_CLASS, '../images/bt-rewFS.png', ACTION_REWIND),
297             getVideoCommand(this.PLAY_PAUSE_BUTTON_CLASS, '../images/bt-playFS.png', ACTION_PLAY),
298             getVideoCommand(this.FAST_FORWARD_BUTTON_CLASS, '../images/bt-ffFS.png', ACTION_FF),
299             getVideoCommand(this.NEXT_BUTTON_CLASS, '../images/bt-nextFS.png', ACTION_NEXT)
300         ];
301     }
302 
303     /**
304     * This method creates a video command object
305     * @private
306     * @param {String} idString The class to give to the button
307     * @param {String} imgUrl Button image url
308     * @param {String} action The action related to the button
309     */
310     function getVideoCommand(idString, imgUrl, action) {
311         return {
312             id: idString,
313             url: imgUrl,
314             action: action
315         };
316     }
317 
318     /**
319     * Format the number of seconds to a string
320     * @private
321     * @param {int} seconds
322     */
323     function formatSeconds(seconds) {
324         var units = (seconds >= 3600) ? 'hours' : 'minutes';
325         return formatTime(seconds, units);
326     }
327 
328     /**
329     * Format the number of seconds and unit to a string
330     * @private
331     * @param {int} seconds
332     * @param {String} unit 'hours' or 'minutes'
333     */
334     function formatTime(seconds, unit) {
335         /**
336         * Format string to add zeros
337         * @private
338         */
339         function padZero(num) {
340             if (num < 10) {
341                 return '0' + num;
342             }
343             return num.toString();
344         }
345 
346         var minutes;
347         switch (unit) {
348             case 'hours':
349                 var hours = Math.floor(seconds / 3600);
350                 minutes = Math.floor(seconds % 3600);
351                 return padZero(hours) + ':' + formatTime(minutes, 'minutes');
352             case 'minutes':
353                 minutes = Math.floor(seconds / 60);
354                 seconds = Math.floor(seconds % 60);
355                 return padZero(minutes) + ':' + formatTime(seconds, 'seconds');
356             case 'seconds':
357                 return padZero(seconds);
358           }
359           return padZero(seconds);
360     }
