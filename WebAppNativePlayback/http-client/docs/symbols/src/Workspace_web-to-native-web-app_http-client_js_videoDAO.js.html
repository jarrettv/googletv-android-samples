168     /**

171     * @param {int} max The max random value
172     */
173     function getRandom(max) {
174         return Math.floor(Math.random() * max);
175     }

177     /**
178     * This method returns a thumbId
179     * @private
180     */
181     function getThumbId() {
182         var max = 14;
183         var num = getRandom(max);
184         if (num === 0) {
185             num = 1;
186         }
187         if (num < 10) {
188             num = '0' + num;
189         }
190         return num.toString();
191     }

194 /**
195  * Returns an array of video category objects
196  */
197  VideoDAO.prototype.getVideoCategories = function() {
198     /**
199     * Helper method to create a category object
200     * @private
201     * @param {String} title The title of the category
202     */
203     function createVideoCategory(title) {
204         return {
205             title: title
206         };
207     }
208 
209      var categoryData = this.getCategoryData();
210      var output = [];
211      for(var i = 0; i < categoryData.length; i++) {
212         output.push(createVideoCategory(categoryData[i].name));
213     }
214     return output;
215 };
216 
217 /**
218 * Returns the array of videos for a given categoryId
219 * @param {int} categoryId
220 */
221 VideoDAO.prototype.getVideoData = function(categoryId) {
222   return this.getCategoryData()[categoryId].videos;
223 }
