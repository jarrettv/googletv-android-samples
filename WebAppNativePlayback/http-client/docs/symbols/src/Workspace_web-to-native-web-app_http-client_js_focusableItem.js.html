 15 /**
 16  * This class is intended to be inherited by all classes which are going to
 17  * handle focusing on elements
 18  *
 19  * @constructor
 20  * @param {DOMElement} domElement The dom element to assosciate with this
 21  * focusable item
 22  */
 23  function FocusableItem(domElement) {
 24     var element = domElement;
 25     var focusState = false;
 26 
 27     /**
 28     * Get the focusable DOM element
 29     */
 30     this.getElement = function () {
 31         return element;
 32     };
 33 
 34     /**
 35     * Set the focus state of this item
 36     * @param {Boolean} isFocused
 37     */
 38     this.setFocusState = function (focus) {
 39         focusState = focus;
 40         this.onFocusStateChange(focus);
 41     };
 42 }
 43 
 44 /**
 45 * This is a callback method when the focus state has changed
 46 * @function
 47 * @param {Boolean} isFocused
 48 */
 49 FocusableItem.prototype.onFocusStateChange = function (isFocused) {
 50     // NOOP
 51 };
 52 
 53 /**
 54 * Callback for when the item is clicked
 55 * @function
 56 */
 57 FocusableItem.prototype.onItemClick = function FocusableItem_onItemClick() {
 58     // NOOP
 59 };
