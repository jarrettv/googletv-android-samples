 15 GridFocusableItem.prototype = new FocusableItem();
 16 
 17 /**
 18  * This is a Focusable Grid Item. It handles the focusing and defocusing of
 19  * the video thumbnails.
 20  *
 21  * @constructor
 22  * @augments FocusableItem
 23  * @param {DOMElement} element The DOM element of the Focusable Item
 24  * @param {String} containerId Container ID of the Focusable item
 25  */
 26 function GridFocusableItem(element, contId) {
 27     FocusableItem.call(this, element);
 28 
 29     var classStates = {
 30         focused: "grid-item-active"
 31     };
 32     var itemClickCallback = null;
 33     var containerId = contId;
 34 
 35     /**
 36     * Set's a callback method for when the item is clicked
 37     * @param {function} callback
 38     */
 39     this.setOnItemClickCallback = function (callback) {
 40         itemClickCallback = callback;
 41     };
 42 
 43     /**
 44     * Get the item click callback
 45     */
 46     this.getOnItemClickCallback = function () {
 47         return itemClickCallback;
 48     };
 49 
 50     /**
 51     * This returns an object with class names associated to each state
 52     */
 53     this.getClassStates = function () {
 54         return classStates;
 55     };
 56 
 57     /**
 58     * This is the container ID of the focusable DOM element
 59     */
 60     this.getContainerId = function () {
 61         return containerId;
 62     };
 63 }
 64 
 65 /**
 66 * Callback when the objects focus state changes
 67 * @param {Boolean} isFocused
 68 */
 69 GridFocusableItem.prototype.onFocusStateChange = function(isFocused) {
 70     var element = this.getElement();
 71     if(isFocused) {
 72         element.addClass(this.getClassStates().focused);
 73 
 74         // Scroll into view
 75         var itemRootParent = $("#"+this.getContainerId());
 76         var elementWrapper = element.parent();
 77 
 78         var offsetAmount = elementWrapper.position().top +
 79         elementWrapper.outerHeight(true) -
 80         itemRootParent.outerHeight(true);
 81 
 82         if(offsetAmount > 0) {
 83             itemRootParent.stop().animate({
 84                 scrollTop: itemRootParent.scrollTop() + offsetAmount
 85             }, 500);
 86         } else if(elementWrapper.position().top < 0) {
 87             itemRootParent.stop().animate({
 88                 scrollTop: itemRootParent.scrollTop() + elementWrapper.position().top
 89             }, 500);
 90         }
 91     } else {
 92         element.removeClass(this.getClassStates().focused);
 93     }
 94 };
 95 
 96 /**
 97 * Callback when the object is clicked
 98 */
 99 GridFocusableItem.prototype.onItemClick = function() {
100     if(this.getOnItemClickCallback()) {
101         this.getOnItemClickCallback()();
102     }
103 };
