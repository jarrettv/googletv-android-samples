 15 /**
 16  * This class is meant to be inherited to control Video playback and then
 17  * each class handles a different method of playback
 18  *
 19  * @constructor
 20  */
 21 function GenericVideoPlayerController() {
 22     var playerCallback = null;
 23     var videoUrl = null;
 24 
 25     /**
 26     * Set a callback method for the player
 27     * @param {function} callback
 28     */
 29     this.setPlayerCallback = function (callback) {
 30         playerCallback = callback;
 31     };
 32 
 33     /**
 34     * Get a player callback
 35     */
 36     this.getPlayerCallback = function () {
 37         return playerCallback;
 38     };
 39 
 40     /**
 41     * Set the video url
 42     * @param {String} url
 43     */
 44     this.setVideoUrl = function (url) {
 45         videoUrl = url;
 46 
 47         this.onVideoUrlChange(url);
 48     };
 49 
 50     /**
 51     * Get the video url
 52     */
 53     this.getVideoUrl = function () {
 54         return videoUrl;
 55     };
 56 }
 57 
 58 /**
 59 * A callback method on a url change
 60 * @function
 61 */
 62 GenericVideoPlayerController.prototype.onVideoUrlChange = function () {
 63     // NOOP
 64 };
 65 
 66 /**
 67 * A method to toggle play / pause
 68 * @function
 69 */
 70 GenericVideoPlayerController.prototype.playPause = function () {
 71     // NOOP
 72 };
 73 
 74 /**
 75 * Method to fast forward
 76 * @function
 77 */
 78 GenericVideoPlayerController.prototype.fastForward = function () {
 79     // NOOP
 80 };
 81 
 82 /**
 83 * Method to rewind
 84 * @function
 85 */
 86 GenericVideoPlayerController.prototype.rewind = function () {
 87     // NOOP
 88 };
 89 
 90 /**
 91 * Callback when the video has loaded
 92 * @function
 93 */
 94 GenericVideoPlayerController.prototype.onLoaded = function () {
 95     // NOOP
 96 };
 97 
 98 /**
 99 * Callback for when the video has ended
100 * @function
101 */
102 GenericVideoPlayerController.prototype.onEnded = function () {
103     // NOOP
104 };
105 
106 /**
107 * Get the number of seconds to skip in the fast forward and rewind
108 * @function
109 */
110 GenericVideoPlayerController.prototype.getSkipSpeedSeconds = function () {
111     return 10;
112 };
