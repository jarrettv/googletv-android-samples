 15 VideoCommandFocusableItem.prototype = new FocusableItem();
 16 
 17  /**
 18  * This class handles the focusing of the video control buttons
 19  *
 20  * @constructor
 21  * @augments FocusableItem
 22  * @param {DOMElement} element Focusable item DOM element
 23  * @param {String} action The action associated with this focusable item
 24  */
 25 function VideoCommandFocusableItem(element, actionString) {
 26 
 27     FocusableItem.call(this, element);
 28 
 29     // TODO: This shouldn't be hard-coded
 30     var classStates = {
 31         focused: 'video-command-focused'
 32     };
 33     var action = actionString;
 34     var itemClickCallback = null;
 35 
 36     /**
 37     * Set the item click callback
 38     * @function
 39     * @param {function} callback
 40     */
 41     this.setOnItemClickCallback = function (callback) {
 42         itemClickCallback = callback;
 43     };
 44 
 45     /**
 46     * Get the item click callback
 47     * @function
 48     */
 49     this.getOnItemClickCallback = function () {
 50         return itemClickCallback;
 51     };
 52 
 53     /**
 54     * Get the action this button should perform
 55     */
 56     this.getAction = function () {
 57         return action;
 58     };
 59 
 60     /**
 61     * Get the class names for each of the focused states
 62     * @function
 63     */
 64     this.getClassStates = function () {
 65         return classStates;
 66     };
 67 }
 68 
 69 /**
 70 * Method called when the focus state of the object changes
 71 * @function
 72 * @param {Boolean} isFocused
 73 */
 74 VideoCommandFocusableItem.prototype.onFocusStateChange = function(isFocused) {
 75     var element = this.getElement();
 76     if(isFocused) {
 77         element.addClass(this.getClassStates().focused);
 78     } else {
 79         element.removeClass(this.getClassStates().focused);
 80     }
 81 };
 82 
 83 /**
 84 * Method called when the object is clicked
 85 * @function
 86 */
 87 VideoCommandFocusableItem.prototype.onItemClick = function() {
 88     if(this.getOnItemClickCallback()) {
 89         this.getOnItemClickCallback()(this.getAction());
 90     }
 91 };
